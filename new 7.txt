Scripts
==============
#!/bin/bash

# Set the threshold (80% in this case)
threshold=80

# Get current disk usage percentage
disk_usage=$(df -h / | awk 'NR==2 {print $5}' | cut -d'%' -f1)

# Check if disk usage exceeds the threshold
if [ "$disk_usage" -gt "$threshold" ]; then
    # Replace the echo statement with your preferred notification method
    echo "Disk space usage is above $threshold% - Current usage: $disk_usage%"
fi


Email alert scripts
==============================
#!/bin/bash

# Set the threshold value
THRESHOLD=36

# Get current disk usage percentage
DISK_USAGE=$(df -h --output=pcent / | tail -n 1 | tr -d ' %')

# Check if disk usage exceeds the threshold
if [ "$DISK_USAGE" -lt "$THRESHOLD" ]; then
    echo "Disk space is below ${THRESHOLD}%. Current usage: ${DISK_USAGE}%." | mail -s "Disk Space Alert" <YOUR_EMAIL>
else
    echo "Disk space is within acceptable limits. Current usage: ${DISK_USAGE}%."
fi
